        // ENHANCED Kitchen Violation Detection
        function checkComprehensiveKitchenViolation(hitter) {
            // Priority 1: Direct volley in kitchen
            if (hitter.inKitchen && ball.height > 0) {
                return {
                    type: 'VOLLEY_IN_KITCHEN',
                    message: 'Kitchen Violation: Cannot volley in kitchen!',
                    severe: true
                };
            }
            
            // Priority 2: Touching kitchen line during volley
            if (!hitter.inKitchen && ball.height > 0) {
                const isPlayerTeam = (hitter === player || hitter === partner);
                const kitchenLine = isPlayerTeam ? KITCHEN_LINE_BOTTOM : KITCHEN_LINE_TOP;
                const distToKitchenLine = Math.abs(hitter.y - kitchenLine);
                
                if (distToKitchenLine < 5) {
                    return {
                        type: 'LINE_FAULT',
                        message: 'Kitchen Violation: Touched line during volley!',
                        severe: true
                    };
                }
            }
            
            // Priority 3: Not re-established after being in kitchen
            if (hitter.wasInKitchen && !hitter.feetEstablished && ball.height > 0) {
                return {
                    type: 'NOT_ESTABLISHED',
                    message: 'Kitchen Violation: Must establish both feet outside!',
                    severe: true
                };
            }
            
            return null;
        }

        const AT_KITCHEN_LINE = Math.abs(hitter.y - KITCHEN_LINE_BOTTOM) < 15;
        const IN_KITCHEN = hitter.inKitchen;
        const LOW_POWER = power < 0.3;
        const FORWARD_ANGLE = Math.abs(angle + Math.PI/2) < Math.PI/3; // Mostly forward toward net
        return (IN_KITCHEN || AT_KITCHEN_LINE) && LOW_POWER && FORWARD_ANGLE;
      }
        
        // ENHANCED Execute dink
        function executeDink(hitter, targetX, targetY) {
            const isPlayerTeam = (hitter === player || hitter === partner);
            
            if (isPlayerTeam) {
                targetY = KITCHEN_LINE_TOP + 15;
            } else {
                targetY = KITCHEN_LINE_BOTTOM - 15;
            }
            
            targetX += (Math.random() - 0.5) * 30;
            
            const dx = targetX - ball.x;
            const dy = targetY - ball.y;
            const dist = Math.sqrt(dx*dx + dy*dy);
            
            ball.vx = (dx / dist) * 80;
            ball.vy = (dy / dist) * 80;
            ball.vz = 60;
            ball.speed = 80;
            
            showMessage('Dink!', hitter.x, hitter.y - 30, '#00BCD4', 40);
            gameState.isDinkRally = true;
            gameState.dinkCount++;
            
            if (hitter === player) {
              updateKitchenPressure(15); // Changed from 10 to 15
            }
            
            return true;
        }
        
        // ENHANCED AI Kitchen Strategy
        function updateAIKitchenStrategy(ai, dt) {
            const isOpponent = (ai === opponent1 || ai === opponent2);
            const ballInTheirHalf = isOpponent ? (ball.y < NET_Y) : (ball.y > NET_Y);
            
            if (!ai.inKitchen && !ai.shouldEnterKitchen) {
                const shouldEnter = shouldAIEnterKitchen(ai, ball);
                if (shouldEnter) {
                    ai.shouldEnterKitchen = true;
                    ai.kitchenTimer = 2.0;
                    ai.preparingDink = true;
                }
            }
            
            if (ai.shouldEnterKitchen && !ai.inKitchen) {
                const kitchenY = isOpponent ? KITCHEN_LINE_TOP - 10 : KITCHEN_LINE_BOTTOM + 10;
                ai.targetY = kitchenY;
                
                if (Math.abs(ai.y - kitchenY) < 10) {
                    ai.inKitchen = true;
                    ai.shouldEnterKitchen = false;
                    ai.wasInKitchen = true;
                    ai.feetEstablished = false;
                }
            }
            
            if (ai.inKitchen && ball.height > 40 && !ai.exitingKitchen) {
                ai.exitingKitchen = true;
                const exitY = isOpponent ? KITCHEN_LINE_TOP - 30 : KITCHEN_LINE_BOTTOM + 30;
                ai.targetY = exitY;
            }
            
            if (ai.exitingKitchen && !ai.inKitchen) {
                ai.exitingKitchen = false;
                ai.establishmentTimer = 0.5;
            }
            
            if (ai.establishmentTimer > 0) {
                ai.establishmentTimer -= dt;
                if (ai.establishmentTimer <= 0) {
                    ai.feetEstablished = true;
                    ai.wasInKitchen = false;
                }
            }
            
            if (ai.kitchenTimer > 0) {
                ai.kitchenTimer -= dt;
                if (ai.kitchenTimer <= 0 && ai.inKitchen) {
                    ai.exitingKitchen = true;
                }
            }
        }
        
        function shouldAIEnterKitchen(ai, ball) {
          const nearNet = Math.abs(ai.y - NET_Y) < 150; // Increased from 100
          const slowBall = ball.speed < 150; // Increased from 120
          const lowBall = ball.height < 50; // Increased from 40
          const ballBouncedNearKitchen = ball.bounces > 0 &&
                Math.abs(ball.y - NET_Y) < KITCHEN_DEPTH + 20;
            
            // More aggressive kitchen play
          return nearNet && (slowBall || lowBall) && Math.random() < ai.skill;
        }

                    if (gameState.kitchenStateTimer <= 0) {

                        updateKitchenButton();
                    }
                    break;

        function isInCorrectServiceBox(x, y, expectedSide, servingFromTop) {
            const isTopCourt = y < NET_Y;

                if (!isTopCourt) {
                    if (y > KITCHEN_LINE_BOTTOM && y < BASELINE_BOTTOM) {
                        const landedLeft = x < COURT_WIDTH / 2;
                        const shouldBeLeft = expectedSide === 'left';
                        return landedLeft === shouldBeLeft;

            } else {
                if (isTopCourt) {
                    if (y > BASELINE_TOP && y < KITCHEN_LINE_TOP) {
                        const landedLeft = x < COURT_WIDTH / 2;
                        const shouldBeLeft = expectedSide === 'left';
                        return landedLeft === shouldBeLeft;
                    }
                }
            }
            
            return false;
        }

        function handleSwipe() {
            if (gameState.waitingForServe && player.isServing) {
                playerServe();
                return;
            }
            
            if (!player.canHit || !gameState.ballInPlay) return;
            
            const dx = touchEnd.x - touchStart.x;
            const dy = touchEnd.y - touchStart.y;
            const distance = Math.sqrt(dx*dx + dy*dy);
            
            if (distance < 30) return;
            
            const power = Math.min(distance / 250, 1);
            const angle = Math.atan2(dy, dx);
            
            // ENHANCED shot type detection
            let shotType = 'normal';
            let speed = 220;
            let arc = 100;
            
            if (isDinkShot(power, angle, player.y)) {
                shotType = 'dink';
                speed = 80;
                arc = 60;
            } else if (power < 0.3 && Math.abs(dy) < 50) {
                shotType = 'drop';
                speed = 100;
                arc = 140;
            } else if (power > 0.7) {
                shotType = 'power';
                speed = 320;
                arc = 80;
            }
            
            // Apply mastery bonuses
            if (gameState.kitchenMastery) {
                if (shotType === 'dink') {
                    speed *= 1.2;
                    arc *= 0.8;
                }
            }
            
            hitBall(player, angle, speed, arc, shotType);
            touchStart = null;
            touchEnd = null;
        }
        
        // ENHANCED hitBall function - YOUR ORIGINAL + KITCHEN ENHANCEMENTS
        function hitBall(hitter, angle, speed, arc, shotType) {
            // Store position for momentum check
            if (ball.height > 0) {
                hitter.volleyPosition = { x: hitter.x, y: hitter.y };
                hitter.momentumTimer = 1.5;
            }
            
            // Check for violations
            const violation = checkComprehensiveKitchenViolation(hitter);
            if (violation) {
                handleKitchenViolation(violation, hitter);
                return;
            }
            
            // YOUR ORIGINAL double bounce check
            if (!gameState.secondBounceComplete && ball.height > 80) {
                if (gameState.consecutiveHits === 0) {
                    showRuleViolation('Must let serve bounce!');
                    faultOccurred(hitter === player || hitter === partner ? 'player' : 'opponent');
                    return;
                } else if (gameState.consecutiveHits === 1) {
                    showRuleViolation('Must let return bounce!');
                    faultOccurred(hitter === player || hitter === partner ? 'player' : 'opponent');
                    return;
                }
            }
            
            // Mark establishment if hitting from ground
            if (hitter.inKitchen && ball.height === 0) {
                hitter.wasInKitchen = false;
                hitter.feetEstablished = true;
            }
            
            // Special handling for dinks
            if (shotType === 'dink' && (hitter.inKitchen || Math.abs(hitter.y - NET_Y) < KITCHEN_DEPTH + 30)) {
                const isPlayerTeam = (hitter === player || hitter === partner);
                const targetX = isPlayerTeam ? 
                    (opponent1.x + opponent2.x) / 2 : 
                    (player.x + partner.x) / 2;
                const targetY = isPlayerTeam ? KITCHEN_LINE_TOP + 20 : KITCHEN_LINE_BOTTOM - 20;
                
                if (executeDink(hitter, targetX, targetY)) {
                    ball.lastHitBy = hitter;
                    ball.lastHitTeam = isPlayerTeam ? 'player' : 'opponent';
                    ball.bounces = 0;
                    ball.bouncePositions = [];
                    ball.inFlight = true;
                    hitter.lastHitTime = Date.now();
                    hitter.canHit = false;
                    gameState.consecutiveHits++;
                    
                    if (gameState.consecutiveHits === 2) {
                        gameState.secondBounceComplete = true;
                    }
                    
                    return;
                }
            }
            
            // YOUR ORIGINAL shot execution
            let targetAngle = angle;
            const isPlayerTeam = (hitter === player || hitter === partner);
            
            if (isPlayerTeam) {
                if (targetAngle > -Math.PI * 0.1) targetAngle = -Math.PI/3;
                targetAngle = Math.max(targetAngle, -Math.PI * 0.75);
                targetAngle = Math.min(targetAngle, -Math.PI * 0.25);
            } else {
                if (targetAngle < Math.PI * 0.1) targetAngle = Math.PI/3;
                targetAngle = Math.max(targetAngle, Math.PI * 0.25);
                targetAngle = Math.min(targetAngle, Math.PI * 0.75);
            }
            
            if (hitter !== player) {
                targetAngle += (Math.random() - 0.5) * 0.3;
                if (Math.random() > hitter.skill) {
                    speed *= 0.8 + Math.random() * 0.4;
                    targetAngle += (Math.random() - 0.5) * 0.5;
                }
            }
            
            ball.vx = Math.cos(targetAngle) * speed;
            ball.vy = Math.sin(targetAngle) * speed;
            ball.vz = arc;
            ball.lastHitBy = hitter;
            ball.lastHitTeam = isPlayerTeam ? 'player' : 'opponent';
            ball.bounces = 0;
            ball.bouncePositions = [];
            ball.inFlight = true;
            ball.speed = speed;
            gameState.bouncesOnCurrentSide = 0;
            
            gameState.isServeInProgress = false;
            gameState.expectedServiceBox = null;
            
            hitter.lastHitTime = Date.now();
            hitter.canHit = false;
            
            gameState.consecutiveHits++;
          gameState.rallyLength++;
            if (gameState.consecutiveHits === 2) {
                gameState.secondBounceComplete = true;
            }
            
            if (ball.lastHitTeam !== gameState.lastHitTeam) {
                gameState.rallyCount++;
                gameState.lastHitTeam = ball.lastHitTeam;
            }
            
            updateUI();
            
            // ENHANCED shot feedback with pressure
            let color = '#4CAF50';
            if (shotType === 'power') {
                color = '#FF5722';
                updateKitchenPressure(5);
            } else if (shotType === 'dink') {
                color = '#00BCD4';
                updateKitchenPressure(15);
            } else if (shotType === 'drop') {
                color = '#FFD600';
                updateKitchenPressure(5);
            }
            
            showMessage(shotType.charAt(0).toUpperCase() + shotType.slice(1) + '!', 
                       hitter.x, hitter.y - 30, color);
            
            // Update kitchen state after hit
            if (gameState.kitchenState === KitchenStates.ACTIVE && hitter === player) {
                if (ball.height > 0) {
                    gameState.kitchenState = KitchenStates.MUST_EXIT;
                    gameState.kitchenStateTimer = 1.0;
                    updateKitchenButton();
                }
            }
        }

        // YOUR ORIGINAL UPDATE WITH KITCHEN ENHANCEMENTS
        function update(dt) {
            if (!gameState.ballInPlay && !gameState.waitingForServe) return;
            
            // Update kitchen systems
            updateKitchenStateMachine(dt);
            updateKitchenMastery(dt);
            
            if (gameState.ballInPlay) {
                // Check momentum violations
                checkMomentumViolations(dt);
                
                // YOUR ORIGINAL ball physics
                const prevY = ball.y;
                
                ball.x += ball.vx * dt;
                ball.y += ball.vy * dt;
                ball.height += ball.vz * dt;
                ball.vz -= GRAVITY * dt;
                
                ball.trail.push({ x: ball.x, y: ball.y, height: ball.height });
                if (ball.trail.length > 25) ball.trail.shift();
                
                const justCrossedNet = (prevY < NET_Y && ball.y >= NET_Y) || 
                                       (prevY > NET_Y && ball.y <= NET_Y);
                if (justCrossedNet) {
                    gameState.bouncesOnCurrentSide = 0;
                    gameState.isDinkRally = false;
                }
                
                // YOUR ORIGINAL bounce logic
                if (ball.height <= 0 && ball.vz < 0) {
                    ball.height = 0;
                    ball.vz = -ball.vz * 0.65;
                    ball.vx *= 0.85;
                    ball.vy *= 0.85;
                    ball.bounces++;
                    gameState.bouncesOnCurrentSide++;
                    
                    const bounceSide = ball.y < NET_Y ? 'opponent' : 'player';
                  
                   // ADD THIS - Check baseline violations
    if (ball.bounces === 1) { // First bounce
        if (bounceSide === 'player') {
            if (ball.y > BASELINE_BOTTOM) {
                showMessage('Long!', ball.x, BASELINE_BOTTOM, '#F44336');
                showRuleViolation('Out - Past baseline!');
                faultOccurred('opponent'); // Opponent's shot was out
                return;
            }
        } else { // opponent side
            if (ball.y < BASELINE_TOP) {
                showMessage('Long!', ball.x, BASELINE_TOP, '#F44336');
                showRuleViolation('Out - Past baseline!');
                faultOccurred('player'); // Player's shot was out
                return;
            }
        }
    }
                  
                    ball.bouncePositions.push({ 
                        x: ball.x, 
                        y: ball.y, 
                        time: Date.now(), 
                        side: bounceSide 
                    });
                    ball.bouncePositions = ball.bouncePositions.filter(b => Date.now() - b.time < 3000);
                    
                    // YOUR ORIGINAL fault checks
                    if (ball.bounces === 1 && !gameState.isServeInProgress) {
                        if ((ball.lastHitTeam === 'player' && bounceSide === 'player') || 
                            (ball.lastHitTeam === 'opponent' && bounceSide === 'opponent')) {
                            showMessage('Own Court!', ball.x, ball.y, '#F44336');
                            showRuleViolation('Fault - Ball bounced on own side!');
                            faultOccurred(ball.lastHitTeam);
                            return;
                        }
                    }
                    
                    // YOUR ORIGINAL service fault checking
                    if (gameState.isServeInProgress && ball.bounces === 1) {
                        const servingFromTop = gameState.servingTeam === 'opponent';
                        
                        const visualBounds = getVisualCourtBounds(ball.y);
                        if (ball.x < visualBounds.left || ball.x > visualBounds.right || 
                            ball.y < 0 || ball.y > COURT_HEIGHT) {
                            showMessage('Service Out!', ball.x, ball.y, '#F44336');
                            showRuleViolation('Service Fault - Out of Bounds!');
                            faultOccurred(gameState.servingTeam);
                            return;
                        }
                        
                        if (servingFromTop) {
                            if (ball.y >= NET_Y && ball.y <= KITCHEN_LINE_BOTTOM) {
                                showMessage('Kitchen Fault!', ball.x, ball.y, '#F44336');
                                showRuleViolation('Service Fault - Ball in Kitchen!');
                                faultOccurred(gameState.servingTeam);
                                return;
                            }
                        } else {
                            if (ball.y <= NET_Y && ball.y >= KITCHEN_LINE_TOP) {
                                showMessage('Kitchen Fault!', ball.x, ball.y, '#F44336');
                                showRuleViolation('Service Fault - Ball in Kitchen!');
                                faultOccurred(gameState.servingTeam);
                                return;
                            }
                        }
                        
                        if (!isInCorrectServiceBox(ball.x, ball.y, 
                                                   gameState.expectedServiceBox, 
                                                   servingFromTop)) {
                            showMessage('Wrong Box!', ball.x, ball.y, '#F44336');
                            showRuleViolation('Service Fault - Wrong Service Box!');
                            faultOccurred(gameState.servingTeam);
                            return;
                        }
                        
                        gameState.isServeInProgress = false;
                        gameState.expectedServiceBox = null;
                        showMessage('Good Serve!', ball.x, ball.y, '#4CAF50', 30);
                    }
                    
                    if (ball.bounces === 1 && gameState.consecutiveHits === 1) {
                        gameState.firstBounceComplete = true;
                    }
                    
                    // ENHANCED Kitchen entry opportunity
                    if (ball.y >= NET_Y && ball.y <= KITCHEN_LINE_BOTTOM && 
                        !gameState.inKitchen && gameState.bouncesOnCurrentSide === 1 &&
                        gameState.kitchenState === KitchenStates.DISABLED) {
                        gameState.kitchenState = KitchenStates.AVAILABLE;
                        gameState.kitchenStateTimer = 2.5;
                        updateKitchenButton();
                    }
                    
                    // YOUR ORIGINAL double bounce fault
                    if (gameState.bouncesOnCurrentSide >= 2) {
                        const faultTeam = bounceSide;
                        showMessage('Double Bounce!', ball.x, ball.y, '#F44336');
                        faultOccurred(faultTeam);
                        return;
                    }

        // YOUR ORIGINAL PLAYER MOVEMENT WITH KITCHEN TRACKING
        function updatePlayerMovement(dt) {
            if (!gameState.ballInPlay) return;
            
            const wasInKitchen = player.inKitchen;
            
            const ballOnOurSide = ball.y > NET_Y;
            
            if (ballOnOurSide && ball.inFlight) {
                const timeToReach = Math.max(0, (player.y - ball.y) / Math.max(ball.vy, 1));
                const predictedX = ball.x + ball.vx * timeToReach * 0.3;
                const inPlayerZone = (player.courtSide === 'right' && predictedX > COURT_WIDTH / 2) || 
                                     (player.courtSide === 'left' && predictedX <= COURT_WIDTH / 2);
                
                if (inPlayerZone || Math.abs(predictedX - player.x) < 80) {
                    const visualBounds = getVisualCourtBounds(player.y);
                    player.targetX = Math.max(visualBounds.left + 15, 
                                            Math.min(predictedX, visualBounds.right - 15));
                    if (!gameState.inKitchen) {
                        const targetY = Math.min(BASELINE_BOTTOM - 20, 
                                                Math.max(KITCHEN_LINE_BOTTOM + 10, ball.y + 15));
                        player.targetY = targetY;
                    }
                }
            } else if (!gameState.inKitchen && !player.isServing) {
                player.targetX = player.courtSide === 'right' ? COURT_WIDTH * 0.7 : COURT_WIDTH * 0.3;
                player.targetY = BASELINE_BOTTOM - SERVICE_LINE_DEPTH;
            }
            
            const moveSpeed = player.speed * (gameState.kitchenMastery ? player.kitchenSpeed : 1.0) * dt;
            const dx = player.targetX - player.x;
            const dy = player.targetY - player.y;
            const dist = Math.sqrt(dx*dx + dy*dy);
            
            if (dist > moveSpeed) {
                player.x += (dx / dist) * moveSpeed;
                player.y += (dy / dist) * moveSpeed;
            } else {
                player.x = player.targetX;
                player.y = player.targetY;
            }
            
            const visualBounds = getVisualCourtBounds(player.y);
            player.x = Math.max(visualBounds.left + 10, Math.min(player.x, visualBounds.right - 10));
            
            if (!player.inKitchen) {
                player.y = Math.max(KITCHEN_LINE_BOTTOM + 5, Math.min(player.y, BASELINE_BOTTOM - 10));
            } else {
                player.y = Math.max(NET_Y + 10, Math.min(player.y, KITCHEN_LINE_BOTTOM - 5));
            }
            
            // Check if entered/exited kitchen zone
            const nowInKitchen = player.inKitchen;
            if (!wasInKitchen && nowInKitchen) {
                player.kitchenEntryTime = Date.now();
                player.wasInKitchen = true;
                player.feetEstablished = false;
            } else if (wasInKitchen && !nowInKitchen && !player.feetEstablished) {
                player.establishmentTimer = 0.5;
            }
            
            const distToBall = Math.sqrt(Math.pow(ball.x - player.x, 2) + 
                                         Math.pow(ball.y - player.y, 2));
            const timeSinceHit = Date.now() - player.lastHitTime;
            player.canHit = distToBall < HIT_DISTANCE && ball.height < 40 && 
                           ball.y > NET_Y && ball.inFlight && timeSinceHit > HIT_COOLDOWN;
        }

                    const arc = 90 + Math.random() * 40;
                    hitBall(partner, angle, speed, arc, 'normal');
                }
            }

            gameState.ballInPlay = false;
            gameState.firstBounceComplete = false;
            gameState.secondBounceComplete = false;
            gameState.consecutiveHits = 0;
            gameState.bouncesOnCurrentSide = 0;

            gameState.isServeInProgress = false;
            gameState.expectedServiceBox = null;
            gameState.isDinkRally = false;
            gameState.dinkCount = 0;
            
            // Reset all players' kitchen states
            [player, partner, opponent1, opponent2].forEach(p => {
                p.inKitchen = false;
                p.wasInKitchen = false;
                p.feetEstablished = true;
                p.momentumTimer = 0;
                p.volleyPosition = null;
                p.establishmentTimer = 0;
            });
            
            updateKitchenButton();
            
          if (gameState.rallyLength > 10) {
        updateKitchenPressure(20); // Reward for long rally
        showMessage('Long Rally Bonus!', COURT_WIDTH/2, COURT_HEIGHT/2, '#4CAF50', 60);
    }
          
            if (gameState.servingTeam === winningTeam) {
                if (winningTeam === 'player') {
                    gameState.playerScore++;
                    switchPlayerSides();
                    updateKitchenPressure(15); // Reward for winning point on serve
                } else {
                    gameState.opponentScore++;
                    switchOpponentSides();
                }
                showMessage('Point!', COURT_WIDTH/2, COURT_HEIGHT/2, '#4CAF50', 90);
            } else {
                if (gameState.isFirstServeOfGame) {
                    gameState.servingTeam = winningTeam;
                    gameState.serverNumber = 1;
                    gameState.isFirstServeOfGame = false;
                    showMessage('Side Out!', COURT_WIDTH/2, COURT_HEIGHT/2, '#FF9800', 90);
                } else if (gameState.serverNumber === 1) {
                    gameState.serverNumber = 2;
                    showMessage('Second Server', COURT_WIDTH/2, COURT_HEIGHT/2, '#FFC107', 90);
                } else {
                    gameState.servingTeam = winningTeam;
                    gameState.serverNumber = 1;
                    showMessage('Side Out!', COURT_WIDTH/2, COURT_HEIGHT/2, '#FF9800', 90);
                }

            gameState.rallyCount = 0;

                }

                resetPositions();
                updateUI();
            }, 1500);
        }

            const isEvenScore = (score % 2) === 0;

            partner.isServing = false;

                    player.isServing = true;
                    player.courtSide = serverSide;
                    partner.courtSide = serverSide === 'right' ? 'left' : 'right';
                    gameState.currentServer = player;
                } else {
                    partner.isServing = true;
                    partner.courtSide = serverSide;
                    player.courtSide = serverSide === 'right' ? 'left' : 'right';
                    gameState.currentServer = partner;

                player.x = player.courtSide === 'right' ? COURT_WIDTH * 0.75 : COURT_WIDTH * 0.25;
                player.y = player.isServing ? BASELINE_BOTTOM - BASELINE_OFFSET : BASELINE_BOTTOM - SERVICE_LINE_DEPTH;

                partner.targetY = partner.y;

                opponent2.courtSide = 'left';
                opponent1.x = COURT_WIDTH * 0.75;

                opponent2.x = COURT_WIDTH * 0.25;

        // YOUR ORIGINAL RENDER FUNCTION WITH KITCHEN ENHANCEMENTS
        function render() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.save();
            
            // Draw background
            const gradient = ctx.createLinearGradient(0, 0, 0, canvas.height);
            gradient.addColorStop(0, '#87CEEB');
            gradient.addColorStop(0.3, '#98D8E8');
            gradient.addColorStop(0.3, '#6B8E6B');
            gradient.addColorStop(1, '#4A5D4E');
            ctx.fillStyle = gradient;
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            drawCourt();
            drawBounceIndicators();
            
            // Draw players (back to front for proper layering)
            if (opponent1.y < opponent2.y) {
                drawPlayer(opponent1, '#CC3333', 'O1');
                drawPlayer(opponent2, '#CC3333', 'O2');
            } else {
                drawPlayer(opponent2, '#CC3333', 'O2');
                drawPlayer(opponent1, '#CC3333', 'O1');
            }
            
            if (partner.y < player.y) {
                drawPlayer(partner, '#2196F3', 'P');
                drawPlayer(player, '#3366CC', 'Y');
            } else {
                drawPlayer(player, '#3366CC', 'Y');
                drawPlayer(partner, '#2196F3', 'P');
            }
            
            if (gameState.ballInPlay) drawBall();
            
            // Draw hit indicator
            if (player.canHit) {
                const screenPos = courtToScreen(player.x, player.y);
                ctx.strokeStyle = '#4CAF50';
                ctx.lineWidth = 3;
                ctx.setLineDash([5, 3]);
                ctx.beginPath();
                ctx.arc(screenPos.x, screenPos.y, HIT_DISTANCE * 0.8, 0, Math.PI * 2);
                ctx.stroke();
                ctx.setLineDash([]);
            }
            
            // Draw establishment indicators
            [player, partner, opponent1, opponent2].forEach(person => {
                if (person.wasInKitchen && !person.feetEstablished) {
                    const screenPos = courtToScreen(person.x, person.y);
                    ctx.strokeStyle = '#FF9800';
                    ctx.lineWidth = 2;
                    ctx.setLineDash([5, 5]);
                    ctx.beginPath();
                    ctx.arc(screenPos.x, screenPos.y, 25, 0, Math.PI * 2);
                    ctx.stroke();
                    ctx.setLineDash([]);
                    
                    if (person === player) {
                        ctx.fillStyle = '#FF9800';
                        ctx.font = 'bold 10px Arial';
                        ctx.textAlign = 'center';
                        ctx.fillText('ESTABLISH', screenPos.x, screenPos.y - 30);
                    }
                }
            });
            
            // Draw messages
            messages.forEach(msg => {
                ctx.fillStyle = msg.color;
                ctx.font = 'bold 14px Arial';
                ctx.textAlign = 'center';
                ctx.fillText(msg.text, msg.x, msg.y);
            });
            
            // Draw mastery effect
            if (gameState.kitchenMastery) {
                const { centerX, courtTop } = getCourtScreenPosition();
                const glowRadius = 150 + Math.sin(Date.now() * 0.005) * 20;
                const gradient = ctx.createRadialGradient(
                    centerX, courtTop + NET_Y * 0.9, 0,
                    centerX, courtTop + NET_Y * 0.9, glowRadius
                );
                gradient.addColorStop(0, 'rgba(255, 215, 0, 0.3)');
                gradient.addColorStop(1, 'rgba(255, 215, 0, 0)');
                ctx.fillStyle = gradient;
                ctx.fillRect(0, 0, canvas.width, canvas.height);
            }
            
            ctx.restore();
        }
        
        // YOUR ORIGINAL DRAW COURT WITH KITCHEN ENHANCEMENTS
        function drawCourt() {
            const { centerX, courtTop } = getCourtScreenPosition();
            
            // Court surface with perspective
            ctx.fillStyle = '#2E7D32';
            ctx.beginPath();
            
            const topLeftX = centerX - (COURT_WIDTH / 2) * PERSPECTIVE_SCALE;
            const topRightX = centerX + (COURT_WIDTH / 2) * PERSPECTIVE_SCALE;
            const bottomLeftX = centerX - COURT_WIDTH / 2;
            const bottomRightX = centerX + COURT_WIDTH / 2;
            
            ctx.moveTo(topLeftX, courtTop);
            ctx.lineTo(topRightX, courtTop);
            ctx.lineTo(bottomRightX, courtTop + COURT_HEIGHT * 0.9);
            ctx.lineTo(bottomLeftX, courtTop + COURT_HEIGHT * 0.9);
            ctx.closePath();
            ctx.fill();
            
            // Add court texture lines
            ctx.strokeStyle = '#236B35';
            ctx.lineWidth = 1;
            for(let i = 0; i <= 10; i++) {
                const y = i / 10;
                const leftX = topLeftX + (bottomLeftX - topLeftX) * y;
                const rightX = topRightX + (bottomRightX - topRightX) * y;
                const screenY = courtTop + COURT_HEIGHT * 0.9 * y;
                
                ctx.beginPath();
                ctx.moveTo(leftX, screenY);
                ctx.lineTo(rightX, screenY);
                ctx.stroke();
            }
            
            // Kitchen zones with dynamic coloring
            const kitchenColor = updateKitchenZoneColor();
            
            const netScreenPos = courtToScreen(COURT_WIDTH/2, NET_Y);
            const kitchenTopScreenPos = courtToScreen(COURT_WIDTH/2, KITCHEN_LINE_TOP);
            const kitchenBottomScreenPos = courtToScreen(COURT_WIDTH/2, KITCHEN_LINE_BOTTOM);
            
            const topKitchenScale = 1 - (1 - PERSPECTIVE_SCALE) * (1 - KITCHEN_LINE_TOP / COURT_HEIGHT);
            const netScale = 1 - (1 - PERSPECTIVE_SCALE) * (1 - NET_Y / COURT_HEIGHT);
            const bottomKitchenScale = 1 - (1 - PERSPECTIVE_SCALE) * (1 - KITCHEN_LINE_BOTTOM / COURT_HEIGHT);
            
            // Top kitchen
            ctx.fillStyle = kitchenColor;
            ctx.beginPath();
            ctx.moveTo(centerX - (COURT_WIDTH/2) * topKitchenScale, kitchenTopScreenPos.y);
            ctx.lineTo(centerX + (COURT_WIDTH/2) * topKitchenScale, kitchenTopScreenPos.y);
            ctx.lineTo(centerX + (COURT_WIDTH/2) * netScale, netScreenPos.y);
            ctx.lineTo(centerX - (COURT_WIDTH/2) * netScale, netScreenPos.y);
            ctx.closePath();
            ctx.fill();
            
            // Bottom kitchen with flash effect
            if (gameState.kitchenFlash && gameState.kitchenFlash.active && 
                gameState.kitchenFlash.side === 'bottom') {
                ctx.fillStyle = `rgba(255, 0, 0, ${gameState.kitchenFlash.opacity})`;
            } else {
                ctx.fillStyle = kitchenColor;
            }
            
            ctx.beginPath();
            ctx.moveTo(centerX - (COURT_WIDTH/2) * netScale, netScreenPos.y);
            ctx.lineTo(centerX + (COURT_WIDTH/2) * netScale, netScreenPos.y);
            ctx.lineTo(centerX + (COURT_WIDTH/2) * bottomKitchenScale, kitchenBottomScreenPos.y);
            ctx.lineTo(centerX - (COURT_WIDTH/2) * bottomKitchenScale, kitchenBottomScreenPos.y);
            ctx.closePath();
            ctx.fill();
            
            // Kitchen labels
            ctx.fillStyle = 'rgba(255, 255, 255, 0.4)';
            ctx.font = 'bold 11px Arial';
            ctx.textAlign = 'center';
            ctx.fillText('KITCHEN', centerX, netScreenPos.y - 30);
            ctx.fillText('KITCHEN', centerX, netScreenPos.y + 35);
            
            // YOUR ORIGINAL COURT LINES
            ctx.strokeStyle = '#FFFFFF';
            ctx.lineWidth = 3;
            
            // Outer boundary
            ctx.beginPath();
            ctx.moveTo(topLeftX, courtTop);
            ctx.lineTo(topRightX, courtTop);
            ctx.lineTo(bottomRightX, courtTop + COURT_HEIGHT * 0.9);
            ctx.lineTo(bottomLeftX, courtTop + COURT_HEIGHT * 0.9);
            ctx.closePath();
            ctx.stroke();
            
            // Baselines
            ctx.lineWidth = 2;

            const baselineBottomPos = courtToScreen(COURT_WIDTH/2, BASELINE_BOTTOM);
            const baselineTopScale = 1 - (1 - PERSPECTIVE_SCALE) * (1 - BASELINE_TOP / COURT_HEIGHT);
            const baselineBottomScale = 1 - (1 - PERSPECTIVE_SCALE) * (1 - BASELINE_BOTTOM / COURT_HEIGHT);
            
            ctx.beginPath();

            ctx.lineTo(centerX + (COURT_WIDTH/2) * baselineTopScale, baselineTopPos.y);
            ctx.moveTo(centerX - (COURT_WIDTH/2) * baselineBottomScale, baselineBottomPos.y);
            ctx.lineTo(centerX + (COURT_WIDTH/2) * baselineBottomScale, baselineBottomPos.y);
            ctx.stroke();
            
            // Kitchen lines
            ctx.strokeStyle = '#FFD700';
            ctx.lineWidth = 2;
            ctx.setLineDash([6, 3]);

            ctx.moveTo(centerX - (COURT_WIDTH/2) * topKitchenScale, kitchenTopScreenPos.y);
            ctx.lineTo(centerX + (COURT_WIDTH/2) * topKitchenScale, kitchenTopScreenPos.y);
            ctx.moveTo(centerX - (COURT_WIDTH/2) * bottomKitchenScale, kitchenBottomScreenPos.y);
            ctx.lineTo(centerX + (COURT_WIDTH/2) * bottomKitchenScale, kitchenBottomScreenPos.y);
            ctx.stroke();
            ctx.setLineDash([]);
            
            // Net
            ctx.strokeStyle = '#000000';
            ctx.lineWidth = 4;
            ctx.beginPath();
            ctx.moveTo(centerX - (COURT_WIDTH/2) * netScale, netScreenPos.y);
            ctx.lineTo(centerX + (COURT_WIDTH/2) * netScale, netScreenPos.y);
            ctx.stroke();

            ctx.lineWidth = 2;
            ctx.beginPath();

            ctx.strokeStyle = 'rgba(255, 255, 255, 0.7)';
            ctx.lineWidth = 2;

            ctx.moveTo(centerX, baselineTopPos.y);

        // YOUR ORIGINAL DRAW FUNCTIONS
        function drawBounceIndicators() {
            ball.bouncePositions.forEach(pos => {
                const age = Date.now() - pos.time;
                const opacity = Math.max(0, 1 - age / 3000);
                ctx.strokeStyle = `rgba(255, 100, 0, ${opacity * 0.6})`;
                ctx.lineWidth = 2;
                const size = 8;
                
                const screenPos = courtToScreen(pos.x, pos.y);
                
                ctx.beginPath();
                ctx.moveTo(screenPos.x - size, screenPos.y - size);
                ctx.lineTo(screenPos.x + size, screenPos.y + size);
                ctx.moveTo(screenPos.x + size, screenPos.y - size);
                ctx.lineTo(screenPos.x - size, screenPos.y + size);
                ctx.stroke();
                
                ctx.beginPath();
                ctx.arc(screenPos.x, screenPos.y, size * 0.7, 0, Math.PI * 2);
                ctx.stroke();
            });
        }
        
        function drawPlayer(playerObj, color, label) {
            const screenPos = courtToScreen(playerObj.x, playerObj.y);
            const perspectiveFactor = 1 - (1 - PERSPECTIVE_SCALE) * (1 - playerObj.y / COURT_HEIGHT);
            const size = playerObj.size * perspectiveFactor;
            
            // Shadow
            ctx.fillStyle = 'rgba(0, 0, 0, 0.3)';
            ctx.beginPath();
            ctx.ellipse(screenPos.x, screenPos.y + 3, size * 0.8, size * 0.4, 0, 0, Math.PI * 2);
            ctx.fill();
            
            // Player circle - with mastery effect
            let playerColor = color;
            if (playerObj === player && gameState.kitchenMastery) {
                playerColor = '#FFD700';
            }
            ctx.fillStyle = playerColor;
            ctx.strokeStyle = 'black';
            ctx.lineWidth = 2;
            ctx.beginPath();
            ctx.arc(screenPos.x, screenPos.y, size, 0, Math.PI * 2);
            ctx.fill();
            ctx.stroke();
            
            // Kitchen indicator
            if (playerObj.inKitchen) {
                ctx.strokeStyle = '#FFD700';
                ctx.lineWidth = 2;
                ctx.setLineDash([3, 2]);
                ctx.beginPath();
                ctx.arc(screenPos.x, screenPos.y, size + 4, 0, Math.PI * 2);
                ctx.stroke();
                ctx.setLineDash([]);
            }
            
            // Court side indicator
            ctx.fillStyle = 'white';
            ctx.font = `bold ${9 * perspectiveFactor}px Arial`;
            ctx.textAlign = 'center';
            ctx.fillText(playerObj.courtSide === 'right' ? 'R' : 'L', screenPos.x, screenPos.y + 3);
            
            // Name label
            ctx.fillStyle = 'white';
            ctx.font = `${10 * perspectiveFactor}px Arial`;
            ctx.fillText(label, screenPos.x, screenPos.y - size - 5);
            
            // Serving indicator
            if (playerObj.isServing) {
                ctx.strokeStyle = '#FFD700';
                ctx.lineWidth = 2;
                ctx.setLineDash([3, 3]);
                ctx.beginPath();
                ctx.arc(screenPos.x, screenPos.y, size + 6, 0, Math.PI * 2);
                ctx.stroke();
                ctx.setLineDash([]);
            }
        }
        
        function drawBall() {
            // Ball trail
            ctx.strokeStyle = 'rgba(255, 215, 0, 0.3)';
            ctx.lineWidth = BALL_RADIUS * 1.5;
            ctx.beginPath();
            ball.trail.forEach((point, i) => {
                const opacity = i / ball.trail.length;
                ctx.globalAlpha = opacity * 0.5;
                const screenPos = courtToScreen(point.x, point.y);
                const visualY = screenPos.y - point.height * 0.15;
                
                if (i === 0) {
                    ctx.moveTo(screenPos.x, visualY);
                } else {
                    ctx.lineTo(screenPos.x, visualY);
                }
            });
            ctx.stroke();
            ctx.globalAlpha = 1;
            
            const ballScreenPos = courtToScreen(ball.x, ball.y);
            const perspectiveFactor = 1 - (1 - PERSPECTIVE_SCALE) * (1 - ball.y / COURT_HEIGHT);
            
            // Shadow
            const shadowSize = (10 + ball.height * 0.08) * perspectiveFactor;
            const shadowOpacity = Math.max(0.15, 0.5 - ball.height * 0.003);
            ctx.fillStyle = `rgba(0, 0, 0, ${shadowOpacity})`;
            ctx.beginPath();
            ctx.ellipse(ballScreenPos.x, ballScreenPos.y, shadowSize, shadowSize * 0.4, 0, 0, Math.PI * 2);
            ctx.fill();
            
            // Ball
            const visualY = ballScreenPos.y - ball.height * 0.18;
            const ballSize = BALL_RADIUS * perspectiveFactor * (1 + ball.height * 0.002);
            
            const gradient = ctx.createRadialGradient(
                ballScreenPos.x - ballSize/3, visualY - ballSize/3, 0,
                ballScreenPos.x, visualY, ballSize
            );
            gradient.addColorStop(0, '#FFFF66');
            gradient.addColorStop(0.6, '#FFD700');
            gradient.addColorStop(1, '#FFA500');
            
            ctx.fillStyle = gradient;
            ctx.strokeStyle = '#FF8C00';
            ctx.lineWidth = 1;
            ctx.beginPath();
            ctx.arc(ballScreenPos.x, visualY, ballSize, 0, Math.PI * 2);
            ctx.fill();
            ctx.stroke();
            
            // Height indicator
            if (ball.height > 25) {
                ctx.strokeStyle = 'rgba(255, 255, 255, 0.2)';
                ctx.lineWidth = 1;
                ctx.setLineDash([2, 2]);
                ctx.beginPath();
                ctx.moveTo(ballScreenPos.x, ballScreenPos.y);
                ctx.lineTo(ballScreenPos.x, visualY);
                ctx.stroke();
                ctx.setLineDash([]);
            }
        }
